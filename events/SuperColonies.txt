#Super Colonies
country_event = {
	id = super_colonies.1
	title = "super_colonies.RHtit1"
	desc = "super_colonies.RHdesc1"
	picture = CONSTITUTION_eventPicture
	
	trigger = {
		is_colonial_nation = no
		OR = {
			#La Plata super colony
			AND = {
				any_subject_country = {
					#ai = yes
					is_colonial_nation = yes
					is_colonial_nation_of = ROOT
					capital_scope = {
						colonial_region = colonial_la_plata
					}
					NOT = { has_country_flag = formed_super_colony }
					#NOT = { has_country_flag = changed_from_colonial_nation }
					#NOT = { liberty_desire = 90 }
					is_at_war = no
				}
				colonial_la_upper_plata = {
					type = all
					country_or_non_sovereign_subject_holds = ROOT
					is_city = yes
				}
				colonial_la_plata = {
					type = all
					country_or_non_sovereign_subject_holds = ROOT
					is_city = yes
				}
			}
		}
	}
	
	mean_time_to_happen = {
		months = 2
	}
	
	option = {
		name = "super_colonies.RHans1"
		#Super colonies
		#La Plata super colony
		every_country = {
			limit = {
				is_colonial_nation = yes
				capital_scope = {
					colonial_region = colonial_la_plata
				}
				is_colonial_nation_of = ROOT
			}
			set_country_flag = formed_super_colony
			#change_tag = LAP
			#remove_non_electors_emperors_from_empire_effect = yes
			#if = {
			#	limit = {
			#		NOT = { government_rank = 2 }
			#	}
			#	set_government_rank = 2
			#}
			if = {
				limit = {
					primary_culture = portugese
				}
				change_primary_culture = brazilian
			}
			colonial_la_plata = {
				#add_permanent_claim = LAP
				add_permanent_claim = PREV
			}
			colonial_la_upper_plata = {
				#add_permanent_claim = LAP
				add_permanent_claim = PREV
			}
			add_country_modifier = {
				name = "centralization_modifier"
				duration = 7300
			}
			add_prestige = 25
			#set_country_flag = changed_from_colonial_nation
			every_province = {
				limit = {
					AND = {
						OR = {
							colonial_region = colonial_la_upper_plata
							colonial_region = colonial_la_plata
						}
						country_or_non_sovereign_subject_holds = ROOT
					}
				}
				remove_core = ROOT
				cede_province = PREV
				add_core = PREV
			}
		}
	}
}
